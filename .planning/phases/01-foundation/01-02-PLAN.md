```csharp
using H.NotifyIcon;
using System.Windows;
using System.Windows.Controls;

namespace TaskbarPet.Services;

public class TrayService : IDisposable
{
    private readonly TaskbarIcon _trayIcon;
    private readonly Window _ownerWindow;

    public TrayService(Window ownerWindow)
    {
        _ownerWindow = ownerWindow;
        _trayIcon = new TaskbarIcon
        {
            IconSource = CreateDefaultIcon(),
            ToolTipText = "Taskbar Pet",
            MenuActivation = PopupActivationMode.LeftOrRightClick
        };
        _trayIcon.ContextMenu = CreateContextMenu();
        _trayIcon.ForceCreate();
    }

    public void HandleTaskbarCreated()
    {
        // Re-create tray icon after Explorer restart
        _trayIcon.ForceCreate();
    }

    private System.Drawing.Icon CreateDefaultIcon()
    {
        // Create a simple default icon programmatically (16x16 with a pet silhouette)
        // For now, use a basic icon — real assets will be added in Phase 2
        using var bitmap = new System.Drawing.Bitmap(16, 16);
        using var graphics = System.Drawing.Graphics.FromImage(bitmap);
        graphics.Clear(System.Drawing.Color.Transparent);
        // Draw a simple circle as placeholder
        graphics.FillEllipse(System.Drawing.Brushes.Gold, 2, 2, 12, 12);
        return System.Drawing.Icon.FromHandle(bitmap.GetHicon());
    }

    private ContextMenu CreateContextMenu()
    {
        var menu = new ContextMenu();
        
        var feedItem = new MenuItem { Header = "Feed" };
        feedItem.Click += (s, e) => OnFeedClicked();
        menu.Items.Add(feedItem);

        var pauseItem = new MenuItem { Header = "Pause" };
        pauseItem.Click += (s, e) => OnPauseClicked();
        menu.Items.Add(pauseItem);

        menu.Items.Add(new Separator());

        var settingsItem = new MenuItem { Header = "Settings" };
        settingsItem.Click += (s, e) => OnSettingsClicked();
        menu.Items.Add(settingsItem);

        var exitItem = new MenuItem { Header = "Exit" };
        exitItem.Click += (s, e) => OnExitClicked();
        menu.Items.Add(exitItem);

        return menu;
    }

    private void OnFeedClicked() { /* Placeholder — wired in Phase 2 */ }
    private void OnPauseClicked() { /* Placeholder — toggle pet animation */ }
    private void OnSettingsClicked() { /* Placeholder — wired in Phase 3 */ }
    private void OnExitClicked() 
    {
        _trayIcon.Dispose();
        Application.Current.Shutdown();
    }

    public void Dispose()
    {
        _trayIcon.Dispose();
    }
}
```

- Add a placeholder `.ico` file to `TaskbarPet/Assets/tray-icon.ico` as a resource
- Or use `H.NotifyIcon.Core.IconHandle` to create an icon from a bitmap without System.Drawing

**MainWindow.xaml.cs**: Add TaskbarCreated message handling via WndProc hook:
```csharp
// Add these fields
private static readonly uint TaskbarCreatedMsg = 
    RegisterWindowMessage("TaskbarCreated");
private HwndSource? _hwndSource;

// In OnSourceInitialized, after WS_EX_NOACTIVATE setup:
protected override void OnSourceInitialized(EventArgs e)
{
    base.OnSourceInitialized(e);
    
    // ... existing WS_EX_NOACTIVATE code ...
    
    // Hook WndProc for TaskbarCreated message
    _hwndSource = HwndSource.FromHwnd(new WindowInteropHelper(this).Handle);
    _hwndSource.AddHook(WndProc);
}

private IntPtr WndProc(IntPtr hwnd, int msg, IntPtr wParam, IntPtr lParam, ref bool handled)
{
    if (msg == TaskbarCreatedMsg)
    {
        // Explorer restarted — tray icon needs re-creation
        // H.NotifyIcon handles this automatically if ForceCreate was called
        // But we explicitly re-set the icon to be safe
        TrayIcon.ForceCreate();
        handled = true;
    }
    return IntPtr.Zero;
}

[DllImport("user32.dll", CharSet = CharSet.Unicode)]
private static extern uint RegisterWindowMessage(string message);

// Event handlers for menu items (placeholders for now):
private void OnFeedClicked(object sender, RoutedEventArgs e) { }
private void OnPauseClicked(object sender, RoutedEventArgs e) { }
private void OnSettingsClicked(object sender, RoutedEventArgs e) { }
private void OnExitClicked(object sender, RoutedEventArgs e)
{
    TrayIcon.Dispose();
    Application.Current.Shutdown();
}

// Clean up on window closing:
protected override void OnClosing(System.ComponentModel.CancelEventArgs e)
{
    TrayIcon.Dispose();
    base.OnClosing(e);
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Tray Icon → Shell | System tray icon communicates with Windows Shell via Shell_NotifyIcon API |
| Context Menu → User Input | Menu items trigger application actions from untrusted user clicks |
| TaskbarCreated Message → Shell | Registered window message from Windows Shell on Explorer restart |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-04 | Tampering | TaskbarCreated message handler | mitigate | Validate message via `RegisterWindowMessage` (system-registered, cannot be spoofed by user processes); only re-create tray icon, no state modification |
| T-01-05 | Denial of Service | Tray icon lifecycle | mitigate | Wrap `ForceCreate()` and `Dispose()` in try/catch; handle `ObjectDisposedException` gracefully to prevent crash on rapid Explorer restart cycles |
| T-01-06 | Information Disclosure | Tray icon tooltip text | accept | Tooltip contains only app name "Taskbar Pet" — no sensitive data exposed |
| T-01-07 | Repudiation | Exit command | mitigate | `OnExitClicked` calls `TrayIcon.Dispose()` before `Application.Current.Shutdown()` ensuring clean icon removal on all exit paths (Pitfall 9) |
</threat_model>
